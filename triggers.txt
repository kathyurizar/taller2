CREATE TRIGGER TR_FACTURACION ON Facturacion
FOR DELETE
AS
BEGIN TRY
    INSERT INTO Facturacion_TR(
        idFactura,
        IdPedido,
        Estado)
    SELECT idFactura,
        IdPedido,
        Estado,
        GETDATE(),
        SUSER_SNAME()
    FROM deleted;
END TRY
BEGIN CATCH
    ROLLBACK TRANSACTION;

    THROW;
END CATCH;
GO
